pod-definition.yaml

apiVersion: v1                    - версия api (у пода v1)     
kind: Pod                        - тип создаваемого объекта
metadata:                        - данные о создаваемом объекте
    name: myapp-pod              - имя пода
    labels:                      - словарь меток (метки позволяют отслеживать созданные объекты)
      app: myapp                   labels может содержать любые пары ключ/значение 
      type: front-end
spec:                            - спецификация. дополнительная информация для k8s о создаваемом объекте
    containers:                  - контейнкры внутри пода (обычно один, но может быть несколько)
      - name: nginx-container    - "-" означает список
        image: nginx
      - command:                 - комманды для контейнера
        - --secure-port=6443
        command: ["some_com"]    - команда при запуске контейнера (аналог ENTRYPOINT в Dockerfile, может его изменять)
        args: ["10"]             - аргументы cmd при запуске контейнера (аналог CMD в Dockerfile, может его изменять)

        command:
          - sleep
          - "1000"

        env:                     - переменные окружения    (env - это массив)
        - name: env_name         - имя переменной          ( - это элемент массива)
          value: env_value       - значение переменной
        - name: env_name_2         
          value: env_value_2
