import lombok.Setter;
import lombok.Getter;

@Setter

    private int height;
    private int length;

    private String color = "black";
    private int numberOfWheels = 4;
    private int numberOfDoors = 5;

    public Car() {

    public Car(int power, int weight, int height, int length) {

        this.weight = weight;
        this.height = height;
        this.length = length;
    }
}
