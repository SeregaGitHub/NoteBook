pod-definition.yaml

apVersion: v1                    - версия api (у пода v1)     
kind: pod                        - тип создаваемого объекта
metadata:                        - данные о создаваемом объекте
    name: myapp-pod              - имя пода
    labels:                      - словарь меток (метки позволяют отслеживать созданные объекты)
      app: myapp                   labels может содержать любые пары ключ/значение 
      type: front-end
spec:                            - спецификация. дополнительная информация для k8s о создаваемом объекте
    containers:                  - контейнкры внутри пода (обычно один, но может быть несколько)
      - name: nginx-container    - "-" означает список
        image: nginx
